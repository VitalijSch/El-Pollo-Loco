    /**
     * Plays a sound if sound is not muted.
     * @param {HTMLAudioElement} sound - The sound to be played.
     * @returns {void}
     */
    playSound(sound) {
        if (!soundMuted) {
            sound.pause();
            sound.currentTime = 0;
            sound.play();
        }
    }
